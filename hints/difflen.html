Here is the whole code. 

def beginning_zeros(a: str) -> int:

    a_num = int(a)
    if not a_num: # case a as all zeros

        return len(a)

    return len(a) - len(str(a_num))

def beginning_zeros(number: str) -> int:

    return len(number) - len(number.lstrip('0'))

Here you iterate over the string, comparing each character with "0" and increasing pre-declared counter in case of equality. 

def beginning_zeros(number: str) -> int:

    result = 0
    for n in number:
        if n != "0":
            break
        result += 1

    return result
